Процедура ПодсчетЧислаДнейСверхуВниз(ДатаНачало, ДатаКонец) 
	РабочиеДни = 0;
	ВыходныеДни = 0; 
	ЧислоНедельВПромежутке = 0;
	ЧислоДнейВПромежутке = ДеньГода(ДатаКонец) - ДеньГода(ДатаНачало) + 1;
	
	Если ЧислоДнейВПромежутке >= 7 ИЛИ ДеньНедели(ДатаНачало) > ДеньНедели(ДатаКонец) Тогда   
		//Первая неделя
		Если ДеньНедели(ДатаНачало) <> 1 тогда   
			ЧислоДнейВПромежутке = ЧислоДнейВПромежутке - (7 - ДеньНедели(ДатаНачало) + 1);
			Если ДеньНедели(ДатаНачало) = 6 ИЛИ ДеньНедели(ДатаНачало) = 7 Тогда
				ВыходныеДни = ВыходныеДни + 8 - ДеньНедели(ДатаНачало);
			Иначе
				ВыходныеДни = ВыходныеДни + 2;
				РабочиеДни = РабочиеДни + (7 - ДеньНедели(ДатаНачало) + 1 - 2); 
			КонецЕсли;
		КонецЕсли;   
		
		//Последняя неделя
		Если ДеньНедели(ДатаКонец) <> 7 тогда   
			ЧислоДнейВПромежутке = ЧислоДнейВПромежутке - ДеньНедели(ДатаКонец);
			Если ДеньНедели(ДатаКонец) = 6 Тогда
				ВыходныеДни = ВыходныеДни + 1;
				РабочиеДни = РабочиеДни + 5;
			Иначе
				РабочиеДни = РабочиеДни + ДеньНедели(ДатаКонец); 
			КонецЕсли;
		КонецЕсли;
		
		//Промежуток
		ЧислоНедельВПромежутке = ЧислоДнейВПромежутке / 7;
		РабочиеДни = РабочиеДни + 5 * ЧислоНедельВПромежутке; 
		ВыходныеДни = ВыходныеДни + 2 * ЧислоНедельВПромежутке; 
	Иначе
		Если ДеньНедели(ДатаКонец) > 5 Тогда
			ВыходныеДни = ВыходныеДни + ДеньНедели(ДатаКонец) - 5;
			РабочиеДни = РабочиеДни + 5;
		Иначе       
			РабочиеДни = РабочиеДни + ДеньНедели(ДатаКонец);	
		КонецЕсли;
		
		Если ДеньНедели(ДатаНачало) > 5 Тогда
			ВыходныеДни = ВыходныеДни - ДеньНедели(ДатаНачало) + 6;
			РабочиеДни = РабочиеДни - 5;
		Иначе       
			РабочиеДни = РабочиеДни - ДеньНедели(ДатаНачало) + 1;	
		КонецЕсли;
	КонецЕсли; 
	
	Сообщить("Рабочих дней: " + РабочиеДни + ". Выходных дней: " + ВыходныеДни);
КонецПроцедуры
